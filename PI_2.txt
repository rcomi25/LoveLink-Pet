#!/usr/bin/env python3
import json
import os
import time
from collections import deque
import paho.mqtt.client as mqtt

BROKER = "127.0.0.1"
PORT = 1883
USER = "username"
PASSWORD = "password"
TOPIC = "test/topic"

STORE_FILE = "/home/pi/mqtt_history.json"
MAX_MESSAGES = 5

last_r = []
last_d = []

pending_r_special = []
pending_d_special = []

SPECIAL_R = {"r_energy", "r_food"}
SPECIAL_D = {"d_energy", "d_food"}

echo_queue = deque()


def trim():
    global last_r, last_d
    last_r = last_r[-MAX_MESSAGES:]
    last_d = last_d[-MAX_MESSAGES:]


def save_store():
    trim()
    with open(STORE_FILE, "w") as f:
        json.dump({
            "last_r": last_r,
            "last_d": last_d,
            "pending_r_special": pending_r_special,
            "pending_d_special": pending_d_special
        }, f)


def load_store():
    global last_r, last_d, pending_r_special, pending_d_special

    if os.path.exists(STORE_FILE):
        try:
            with open(STORE_FILE, "r") as f:
                data = json.load(f)

            last_r = data.get("last_r", [])[-MAX_MESSAGES:]
            last_d = data.get("last_d", [])[-MAX_MESSAGES:]

            pending_r_special = [m for m in data.get("pending_r_special", []) if m in SPECIAL_R]
            pending_d_special = [m for m in data.get("pending_d_special", []) if m in SPECIAL_D]

        except Exception:
            last_r = []
            last_d = []
            pending_r_special = []
            pending_d_special = []

    save_store()


def add_unique(lst, msg):
    if msg not in lst:
        lst.append(msg)


def is_echo(payload):
    global echo_queue

    for i, msg in enumerate(echo_queue):
        if msg == payload:
            del echo_queue[i]
            return True

    return False


def publish_bridge(client, msg):
    echo_queue.append(msg)
    print("TX:", msg, flush=True)
    client.publish(TOPIC, msg, qos=0, retain=False)
    time.sleep(0.25)


def send_messages(client, messages):
    for msg in list(messages):
        publish_bridge(client, msg)


def on_connect(client, userdata, flags, rc):
    if rc == 0:
        print("Connected", flush=True)
        client.subscribe(TOPIC, qos=0)
    else:
        print("Connection failed:", rc, flush=True)


def on_message(client, userdata, message):
    global last_r, last_d, pending_r_special, pending_d_special

    payload = message.payload.decode("utf-8", errors="ignore").strip()
    print("RX:", payload, flush=True)

    if is_echo(payload):
        print("Ignored bridge echo:", payload, flush=True)
        return

    if payload == "r_on":
        print("r_on -> send last d_ + one-shot d_energy/d_food", flush=True)

        send_messages(client, last_d[-MAX_MESSAGES:])

        if pending_d_special:
            send_messages(client, pending_d_special)
            pending_d_special = []
            save_store()

        return

    if payload == "d_on":
        print("d_on -> send last r_ + one-shot r_energy/r_food", flush=True)

        send_messages(client, last_r[-MAX_MESSAGES:])

        if pending_r_special:
            send_messages(client, pending_r_special)
            pending_r_special = []
            save_store()

        return

    if payload in SPECIAL_R:
        add_unique(pending_r_special, payload)
        save_store()
        print("Saved pending R special:", pending_r_special, flush=True)
        return

    if payload in SPECIAL_D:
        add_unique(pending_d_special, payload)
        save_store()
        print("Saved pending D special:", pending_d_special, flush=True)
        return

    if payload.startswith("r_"):
        if not last_r or last_r[-1] != payload:
            last_r.append(payload)
            save_store()
        print("Saved R:", last_r, flush=True)
        return

    if payload.startswith("d_"):
        if not last_d or last_d[-1] != payload:
            last_d.append(payload)
            save_store()
        print("Saved D:", last_d, flush=True)
        return


load_store()

client = mqtt.Client(client_id="mqtt-history-bridge-main")
client.username_pw_set(USER, PASSWORD)
client.on_connect = on_connect
client.on_message = on_message

client.connect(BROKER, PORT, 60)
client.loop_forever()
